      def initialize(options = {})
        options.each do |key, value|
          self.send("#{key}=", value)
        end
        
        @headers ||= {}
        @http_service ||= GData::HTTP::DefaultService
        @version ||= '2'
        @source ||= 'AnonymousApp'
      end
