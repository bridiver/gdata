      def initialize(options = {})
        options[:clientlogin_service] ||= 'youtube'
        options[:clientlogin_url] ||= 'https://www.google.com/youtube/accounts/ClientLogin'
        options[:authsub_scope] ||= 'http://gdata.youtube.com'
        options[:version] ||= '2'
        super(options)
      end
