      def initialize(entry, file, file_mime_type)
        @boundary = "END_OF_PART_#{rand(64000)}"
        entry = wrap_entry(entry, file_mime_type)
        closing_boundary = MimeBodyString.new("\r\n--#{@boundary}--")
        @parts = [entry, file, closing_boundary]
        @current_part = 0
      end
